      // C++
      unsigned ll lattice(int m, int n){
        int tot = m + n;
        int val = min(m, n);
        if (val == 0) return 1;
        unsigned ll result = tot;
        for (int i = 2; i <= val; i++){
            result *= (tot-i+1);
            result /= i;
        }
        return result;
    }

      // C++
      string intToEnglish(int num){
        // no pattern here
        if (num < 20){
            switch(num){
                case  0: return "Zero";
                case  1: return "One";
                case  2: return "Two";
                case  3: return "Three";
                case  4: return "Four";
                case  5: return "Five";
                case  6: return "Six";
                case  7: return "Seven";
                case  8: return "Eight";
                case  9: return "Nine";
                case 10: return "Ten";
                case 11: return "Eleven";
                case 12: return "Twelve";
                case 13: return "Thirteen";
                case 14: return "Fourteen";
                case 15: return "Fifteen";
                case 16: return "Sixteen";
                case 17: return "Seventeen";
                case 18: return "Eighteen";
                case 19: return "Nineteen";
                default: break;
            }
        }
        // pattern here tens the same
        // ones/units we feed back in
        else if (num > 19 && num < 100){
            int ones = num % 10;
            string suffix = ones != 0 ? intToEnglish(ones) : "";
            int tens = num / 10;
            switch(tens){
                case 2: return "Twenty" + suffix;
                case 3: return "Thirty" + suffix;
                case 4: return "Forty" + suffix;
                case 5: return "Fifty" + suffix;
                case 6: return "Sixty" + suffix;
                case 7: return "Seventy" + suffix;
                case 8: return "Eighty" + suffix;
                case 9: return "Ninety" + suffix;
                default: break;
            }
        }
        // one thousand case can implement 1000+ later
        // if needed but for now too much work
        else if (num == 1000){
            return "OneThousand";
        }
        // feed hundreds in add english "Hundred"
        // add an and if the tens isnt 00
        // then add the suffix
        int tens = num % 100;
        string suffix = tens != 0 ? intToEnglish(tens) : "";;
        int hundreds = num / 100;
        string n = tens != 0 ? "and" : "";
        return intToEnglish(hundreds) + "Hundred" + n + suffix;
    }

        // https://en.wikipedia.org/wiki/Determination_of_the_day_of_the_week
        // using Guass's algorithm
        int t[] = {0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4};

        return 0 == (( y + y/4 - y/100 + y/400 + t[m-1] + 1) % 7);

      import math
      primes = [2, 3, 5, 7]
      for number in range(9, 1000000, 2):
          half = int(math.sqrt(number)) + 1
          broken = False
          for check in primes:
              if check > half:
                  break
              elif number % check == 0:
                  broken = True
                  break

              primes.append(number)

      starts = ['2', '3', '5', '7']
      mids = ['1', '3', '7', '9']
      ends = ['3', '7']
      def is_allowed(number):
          number = str(number)
          if len(number) == 1:
              return False
          if number[0] not in starts:
              return False
          if number[-1] not in ends:
              return False
          return not any(c not in mids for c in number[1:-1])

      test_primes = list(filter(is_allowed, primes))
      primes = list(map(lambda num : str(num), primes))
      test_primes = list(map(lambda num : str(num), test_primes))

      def is_truncatable(prime):
          length = len(prime)
          for i in range(1, length):
              if prime[i:] not in primes:
                  break
              if prime[:length - i] not in primes:
                  break
          else:
              return True
          return False

      prime_sum = 0
      c = 0
      for prime in test_primes:
          length = len(prime)
          if is_truncatable(prime):
              print(f"{prime} - is truncatable")
              c += 1
              prime_sum += int(prime)
      print(c)
      print(prime_sum)
